{

        for(i=1;i<=7;i++)
        {
                int a,b;
                scanf("%d%d",&a,&b);
                hp[i]=a+b;
        }
        for(i=1;i<=6;i++)
        {
                if(hp[i]>8) hp[i]-=8;
                else hp[i]=0;
     //           hp[i+1]+=hp[i];
        }
        if(hp[7]>8) hp[7]-=9;
        else hp[7]=0;
        int t=0,h=0;
        for(i=1;i<=7;i++)
        {
               if(hp[i]>h)
               {
                        t=i;
                        h=hp[i];
               }
        }
        printf("%d",t);
//        while(1);

}
