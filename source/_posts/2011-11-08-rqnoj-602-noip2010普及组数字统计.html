//对对，我知道那个tj函数不用这样，但是这样代码就有重用的可能性了。
//水题，毕竟后来发现这是普及组的。
#include <stdio.h>
int sta[10]={0};
void tj(int k)
{
        if(k<10) sta[k]++;
        else
        {
                sta[k%10]++;
                tj(k/10);
        }
}
int main()
{
        int l,r;
        scanf("%d%d",&l,&r);
        int i;
        for(i=l;i<=r;i++) tj(i);
        printf("%d",sta[2]);
        return 0;
}
