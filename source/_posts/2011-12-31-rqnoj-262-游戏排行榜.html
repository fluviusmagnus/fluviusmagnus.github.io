//稍微有点算是水题。吧。
#include <stdio.h>
#include <limits.h>
int main()
{
        unsigned int list[51]={0};
        list[0]=INT_MAX;
        int p,s;
        unsigned int n;
        scanf("%d%d%d",&p,&s,&n);
        int i;
        for(i=1;i<=s;i++)
                scanf("%d",&list[i]);
        for(i=1;i<=p;i++)
                if(n>=list[i]&&n>list[p]&&n<list[i-1])
                {
                        printf("%d",i);
                        return 0;
                }
        printf("-1");
        return 0;
}
