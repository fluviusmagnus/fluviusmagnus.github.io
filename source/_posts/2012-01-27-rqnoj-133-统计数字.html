//第一次拙劣地使用了STL。以此纪念。
#include <stdio.h>
#include <set>
using namespace std;
int main()
{
	set<int> num;
	multiset<int> time;
	int n,i,t;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&t);
		num.insert(t);
		time.insert(t);
	}
	set<int>::iterator it;
	for(it=num.begin();it!=num.end();it++)
		printf("%d %d\n",*it,time.count(*it));
	return 0;
}
