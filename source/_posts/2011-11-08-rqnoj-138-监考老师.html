#include <stdio.h>
int n,m;
int map[101][101]={0};
int ans[101][101]={0};
int get(int x,int y)
{
        int sum,i;
        sum=0;
        for(i=1;i<=m;i++) sum+=map[x][i];
        for(i=1;i<=n;i++) sum+=map[i][y];
        return sum;
}
int main()
{
        int i,r;
        scanf("%d%d",&n,&m);
        for(i=1;i<=n;i++)
                for(r=1;r<=m;r++)
                        scanf("%d",&map[i][r]);
        int max=0;
        for(i=1;i<=n;i++)
                for(r=1;r<=m;r++)
                        if(get(i,r)>max) max=get(i,r);
        printf("%d",max);
        return 0;
}
