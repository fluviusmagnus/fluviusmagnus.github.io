//其实就是最笨的办法。想太多就死了。
#include <stdio.h>
int main()
{
        int n;
        int f[101];
        scanf("%d",&n);
        int i,sum=0;
        for(i=1;i<=n;i++)
        {
                scanf("%d",&f[i]);
                sum+=f[i];
        }
        sum/=n;
        int count=0;
        for(i=1;i<n;i++)
        {
                if(f[i]!=sum)
                {
                        f[i+1]+=f[i]-sum;
                        f[i]=sum;
                        count++;
                }
        }
        printf("%d",count);
        return 0;
}
