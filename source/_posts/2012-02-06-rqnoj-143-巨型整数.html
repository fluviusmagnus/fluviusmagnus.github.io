#include <stdio.h>
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
int main()
{
	string s;
	char in[20001];
	scanf("%s",in);
	s=in;
	reverse(s.begin(),s.end());
	string::iterator it=s.begin();
//此处需要注意，输入100须输出1。
	while(*it=='0')
		s.erase(it);
	cout<<s<<endl;
	long int sum=0;
	for(it=s.begin();it!=s.end();it++)
		sum+=(*it-'0');
	cout<<sum;
	return 0;
}
