//扫一遍好了。规模不大。如果很大就要换种方法了。
#include <stdio.h>
float a,b,c,d;
float f(float x)
{
        return a*x*x*x+b*x*x+c*x+d;
}
int main()
{
        float x;
        int sum=0;
        scanf("%f%f%f%f",&a,&b,&c,&d);
        x=-100;
        while(sum!=3)
        {
                if(f(x)*f(x+0.01)<=0)
                {
                        printf("%0.2f ",x+0.01);
                        sum++;

                }
                x+=0.01;
        }
        return 0;
}
