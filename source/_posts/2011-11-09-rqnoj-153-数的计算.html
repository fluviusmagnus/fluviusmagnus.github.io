#include <stdio.h>
int ff(int now)
{
        if(now==0) return 1;
        else
        {
                int i,sum=0;
                for(i=now/2;i>=0;i--)
                        sum+=ff(i);
                return sum;
        }
}
int main()
{
        int n;
        scanf("%d",&n);
        printf("%d",ff(n));
        return 0;
}
