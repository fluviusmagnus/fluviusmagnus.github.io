//特别注意要强制类型转换。
//当然也可以不用。这是另一份参考：http://sfiction.blog.163.com/blog/static/199404010201110835749863/
#include <stdio.h>
int main()
{
        double sn=0;
        int k;
        int sum=0;
        scanf("%d",&k);
        while(sn<=k)
        {
                sum++;
                sn+=1/(double)sum;
        }
        printf("%d",sum);
        return 0;
}
