    for(r=0;r<=t;r++) if(r<c[1]) f[1][r]=0; else f[1][r]=v[1];
    for(i=2;i<=m;i++)
        for(r=0;r<=t;r++) 
            if(r>=c[i])
                f[i][r]=max(f[i-1][r],f[i-1][r-c[i]]+v[i]);
            else
                f[i][r]=f[i-1][r];
    printf("%d",f[m][t]);

//关键！为保证本次所需要的未用数据不改变，令r递减。

//做着做着发现分子分母反了。所以输出时反一下。
#include <stdio.h>
int main()
{
        int n;
        int c=1;
        int zi=1,mu=1;
        int i=1;
        int pos;
        scanf("%d",&n);
        for(c=1;c<=n;c++)
                if((c+1)*c/2>=n&&c*(c-1)/2<n) break;
        if(c%2==1)
        {
                pos=(c-1)*c/2;
                zi=n-pos;
                mu=c+1-zi;
        }
        else
        {
                pos=(c-1)*c/2;
                zi=n-pos+c-1;
                mu=c+1-zi;
        }
        printf("%d/%d",mu,zi);
        return 0;
}
