RQNOJ 144 伯虎点秋香

Nov 9th, 2011 | 评论 

//此题应当注意字符串的操作。
//确保字符串函数不要用错！
#include <stdio.h>
#include <string.h>
void swap(char* a,char* b)
{
        char temp[1001];
        strcpy(temp,a);
        strcpy(a,b);
        strcpy(b,temp);
}
int main()
{
        int n;
        char name[1001][1001];
        char nnn[1001];
        scanf("%d",&n);
        int k;
        scanf("%d",&k);
        int i,r;
        for(i=1;i<=n;i++)
        {
            scanf("%s",&name[i][0]);
        }
        for(i=1;i<=n;i++)
                for(r=n;r>=i+1;r--)
                        if(strcmp(&name[i][0],&name[r][0])>0) swap(&name[i][0],&name[r][0]);
        printf("%s",&name[k][0]);
        return 0;
}
