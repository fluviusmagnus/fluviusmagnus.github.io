RQNOJ 394 [NOIP2008]火柴棒等式

Nov 10th, 2011 | 评论 

//很忐忑地把1000以内的爆搜交上去了。结果万幸。
#include <stdio.h>
int stick[10]={6,2,5,5,4,5,6,3,7,6};
int tj(int num)
{
        if(num<10) return stick[num];
        else return stick[num%10]+tj(num/10);
}
int main()
{
        int i,r;
        int n;
        scanf("%d",&n);
        int count=0;
        for(i=0;i<=1000;i++)
                for(r=0;r<=1000;r++)
                        if(tj(i)+tj(r)+tj(i+r)==n-4) count++;
        printf("%d",count);
        return 0;
}
